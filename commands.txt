import requests
import csv
import subprocess
import io

url = "https://feodotracker.abuse.ch/downloads/ipblocklist.csv"

print("[+] Downloading IP blocklist...")
response = requests.get(url)

if response.status_code != 200:
    print("[-] Failed to download list")
    exit()

print("[+] Removing old firewall rule (if exists)...")
subprocess.run([
    "powershell",
    "-Command",
    'netsh advfirewall firewall delete rule name="BadIPs"'
], stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL)

print("[+] Parsing CSV...")

lines = [
    line for line in response.text.splitlines()
    if not line.startswith("#") and line.strip() != ""
]

csv_reader = csv.reader(lines)

blocked_count = 0

for row in csv_reader:
    ip = row[0].strip()

    if ip.lower() == "dst_ip":
        continue

    rule_out = f'netsh advfirewall firewall add rule name="BadIPs" dir=out action=block remoteip={ip}'
    rule_in  = f'netsh advfirewall firewall add rule name="BadIPs" dir=in action=block remoteip={ip}'

    subprocess.run(["powershell", "-Command", rule_out], stdout=subprocess.DEVNULL)
    subprocess.run(["powershell", "-Command", rule_in], stdout=subprocess.DEVNULL)

    blocked_count += 1
    print(f"[+] Blocked {ip}")

print(f"\n Done. Total IPs blocked: {blocked_count}")

import java.util.*;
class password{
public static String checkPasswordStrength(String password){
    if(password.length()<8){
        return "Weak: Password must be at least 8 characters long.";
    }
    boolean hasDigit=false;
    boolean hasUpper=false;
    boolean hasLower=false;
    boolean hasSpecial = false;
    for(char ch:password.toCharArray()){
        if(Character.isDigit(ch)){
            hasDigit=true;
        }
        if(Character.isUpperCase(ch)){
            hasUpper=true;
        }
        if(Character.isLowerCase(ch)){
            hasLower=true;
        }
    if(!Character.isLetterOrDigit(ch)){
        hasSpecial = true;
        }

    }
    if(!hasDigit){
        return "Weak: Password must contain at least one digit.";
    }
    if(!hasUpper){
        return "Weak: Password must contain at least one uppercase letter.";
    }
    if(!hasLower){
        return "Weak: Password must contain at least one lowercase letter.";
    } 
    if(!hasSpecial){
    return "Weak: Password must contain at least one special character.";
}

    return "Strong password.";
}
public static void main(String[] args){
    Scanner s=new Scanner(System.in);
    System.out.println("Welcome to the Password Strength Checker");
        while (true) {
            System.out.print("\nEnter your password (or type exit to quit): ");
            String password = s.nextLine();

            if (password.equalsIgnoreCase("exit")) {
                System.out.println("Thank you for using the Password Strength Checker! Goodbye!");
                break;
            }
            String result = checkPasswordStrength(password);
            System.out.println(result);
        } 
}
